#+title: Hello World Example


To run, in two separate terminals, do:

#+begin_src fish :results raw
cargo run \
  --bin 'server' \
  -- \
  --address '127.0.0.1:4030'
#+end_src

#+RESULTS:
Starting up server at 127.0.0.1:4030
response: Hello, World!

and

#+begin_src fish :results raw
cargo run \
  --bin 'client' \
  -- \
  --address '127.0.0.1:4030' \
  --name "World"
#+end_src

#+RESULTS:
received: Hello, World!
